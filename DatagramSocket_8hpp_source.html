707 template <typename T> T read()
708 {
709 static_assert(std::is_trivially_copyable_v<T>,
710 "DatagramSocket::read<T>() only supports trivially copyable types");
711
712 if (!_isConnected)
713 throw SocketException("DatagramSocket is not connected.");
714
715 T value;
716 const int n = ::recv(getSocketFd(), reinterpret_cast<char*>(&value), sizeof(T), 0);
717
718 if (n == SOCKET_ERROR)
719 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
720 if (n == 0)
721 throw SocketException("Connection closed by remote host.");
722 if (n != sizeof(T))
723 throw SocketException("Partial read: did not receive a complete T object.");
724 return value;

736 {
737 static_assert(std::is_trivially_copyable_v<T>, "recvFrom<T>() only supports trivially copyable types");
738

741 T value;
742 const auto n = ::recvfrom(getSocketFd(), reinterpret_cast<char*>(&value), sizeof(T), 0,
743 reinterpret_cast<sockaddr*>(&srcAddr), &srcLen);
744 if (n == SOCKET_ERROR)

746 if (n == 0)
747 throw SocketException("Connection closed by remote host.");
748
749 // Get sender's IP:port
