1375 {
1376 try

1378 internal::sendExactTo(getSocketFd(), buffer.data(), buffer.size(), ai->ai_addr,

1671 template <typename T, std::enable_if_t<detail::is_dynamic_buffer_v<T>, int> = 0>
1672 [[nodiscard]] T read(const DatagramReadOptions& opts = {},
1673 std::size_t minCapacity = DefaultDatagramReceiveSize) const
1674 {
1675 if (minCapacity == 0)
1676 minCapacity = 1; // ensure we can receive zero-length datagrams cleanly
1677
1678 std::size_t capacity = (std::min) (minCapacity, MaxDatagramPayloadSafe);
1679
1680 // Optional preflight probe to choose a better starting capacity
1681 if (opts.mode != DatagramReceiveMode::NoPreflight)
1682 {
1683 try
1684 {
1685 if (const std::size_t exact = internal::nextDatagramSize(getSocketFd()); exact > 0)
1686 {
1687 const std::size_t clamped = (std::min) (exact, MaxDatagramPayloadSafe);
1688 capacity = (std::max) (capacity, clamped);
1689 }
1690 }
1691 catch (const SocketException&)
1692 {
1693 // Graceful degradation: keep current capacity
1694 }
1695 }
1696
1697 T out;
1698 out.resize(capacity);
1699
1700 // Delegate to the backbone — ensures consistent error/timeout/truncation handling
1701 // Shrink to actual bytes received (never null-terminate automatically)
1702 if (auto res = readInto(static_cast<void*>(out.data()), out.size(), opts); res.bytes < out.size())
1703 out.resize(res.bytes);
1704
1705 return out;
1706 }
