354 template <typename T> T read()
355 {
356 static_assert(std::is_trivially_copyable_v<T>,
357 "DatagramSocket::read<T>() only supports trivially copyable types");
358
359 if (!_isConnected)
360 throw SocketException("DatagramSocket is not connected.");
361
362 T value;
363 const int n = ::recv(getSocketFd(), reinterpret_cast<char*>(&value), sizeof(T), 0);
364
365 if (n == SOCKET_ERROR)
366 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
367 if (n == 0)
368 throw SocketException("Connection closed by remote host.");
369 if (n != sizeof(T))
370 throw SocketException("Partial read: did not receive a complete T object.");
371 return value;
372 }
