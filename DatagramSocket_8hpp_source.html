284 template <typename T> T read()
285 {
286 static_assert(std::is_trivially_copyable_v<T>,
287 "DatagramSocket::read<T>() only supports trivially copyable types");
288
289 if (!_isConnected)
290 throw SocketException("DatagramSocket is not connected.");
291
292 T value;
293 const int n = ::recv(_sockFd, reinterpret_cast<char*>(&value), sizeof(T), 0);
294
295 if (n == SOCKET_ERROR)
296 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
297 if (n == 0)
298 throw SocketException("Connection closed by remote host.");
299 if (n != sizeof(T))
300 throw SocketException("Partial read: did not receive a complete T object.");
301 return value;
302 }

312 template <typename T> T recvFrom(std::string* senderAddr, Port* senderPort)
313 {
