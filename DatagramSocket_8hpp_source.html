673 template <typename T> T read()
674 {
675 static_assert(std::is_trivially_copyable_v<T>,
676 "DatagramSocket::read<T>() only supports trivially copyable types");
677
678 if (!_isConnected)
679 throw SocketException("DatagramSocket is not connected.");
680
681 T value;
682 const int n = ::recv(getSocketFd(), reinterpret_cast<char*>(&value), sizeof(T), 0);
683
684 if (n == SOCKET_ERROR)
685 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
686 if (n == 0)
687 throw SocketException("Connection closed by remote host.");
688 if (n != sizeof(T))
689 throw SocketException("Partial read: did not receive a complete T object.");
690 return value;
691 }

701 template <typename T> T recvFrom(std::string* senderAddr, Port* senderPort)
702 {
703 static_assert(std::is_trivially_copyable_v<T>, "recvFrom<T>() only supports trivially copyable types");
704
705 sockaddr_storage srcAddr{};
706 socklen_t srcLen = sizeof(srcAddr);
707 T value;
708 const auto n = ::recvfrom(getSocketFd(), reinterpret_cast<char*>(&value), sizeof(T), 0,
709 reinterpret_cast<sockaddr*>(&srcAddr), &srcLen);
710 if (n == SOCKET_ERROR)
711 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
712 if (n == 0)
713 throw SocketException("Connection closed by remote host.");
714
715 // Get sender's IP:port
716 char hostBuf[NI_MAXHOST], portBuf[NI_MAXSERV];
717 if (getnameinfo(reinterpret_cast<sockaddr*>(&srcAddr), srcLen, hostBuf, sizeof(hostBuf), portBuf,
718 sizeof(portBuf), NI_NUMERICHOST | NI_NUMERICSERV) == 0)
719 {
720 if (senderAddr)
721 *senderAddr = hostBuf;
722 if (senderPort)
723 *senderPort = static_cast<Port>(std::stoul(portBuf));
724 }
725 else
