257 template <typename T> T read()
258 {
259 static_assert(std::is_trivially_copyable_v<T>,
260 "DatagramSocket::read<T>() only supports trivially copyable types");
261
262 if (!_isConnected)
263 throw SocketException("DatagramSocket is not connected.");
264
265 T value;
266 const int n = ::recv(_sockFd, reinterpret_cast<char*>(&value), sizeof(T), 0);
267
268 if (n == SOCKET_ERROR)
269 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
270 if (n == 0)
271 throw SocketException("Connection closed by remote host.");
272 if (n != sizeof(T))
273 throw SocketException("Partial read: did not receive a complete T object.");
274 return value;

286 {
