913 const int socktype, const int protocol, const int flags = 0)

1245inline std::size_t nextDatagramSize(SOCKET fd) noexcept
1246{
1247#if defined(_WIN32)
1248 u_long pending = 0;
1249 return (ioctlsocket(fd, FIONREAD, &pending) == 0 && pending > 0) ? static_cast<std::size_t>(pending) : 0;
1250#else
1251 int pending = 0;
1252 if (ioctl(fd, FIONREAD, &pending) == 0 && pending > 0)
1253 return static_cast<std::size_t>(pending);
1254
1255 // POSIX: MSG_PEEK|MSG_TRUNC returns exact datagram length without consuming it.
1256 sockaddr_storage tmp{};
1257 socklen_t tmpLen = static_cast<socklen_t>(sizeof(tmp));
1258 const ssize_t peekN = ::recvfrom(fd, nullptr, 0, MSG_PEEK | MSG_TRUNC, reinterpret_cast<sockaddr*>(&tmp), &tmpLen);
1259 return (peekN > 0) ? static_cast<std::size_t>(peekN) : 0;
1260#endif
1261}

1383inline void resolveNumericHostPort(const sockaddr* sa, const socklen_t len, std::string& host, Port& port)
1384{
1385 char hostBuf[NI_MAXHOST]{};
1386 char servBuf[NI_MAXSERV]{};
1387
1388 int flags = NI_NUMERICHOST | NI_NUMERICSERV;
1389#ifdef NI_NUMERICSCOPE
1390 flags |= NI_NUMERICSCOPE; // ensure scope id stays numeric on IPv6 link-local
1391#endif
1392
1393 if (const int ret = ::getnameinfo(sa, len, hostBuf, sizeof(hostBuf), servBuf, sizeof(servBuf), flags); ret != 0)
1394 {
1395 throw SocketException(
1396#ifdef _WIN32
1397 GetSocketError(), SocketErrorMessageWrap(GetSocketError(), true));
1398#else
1399 ret, SocketErrorMessageWrap(ret, true));
1400#endif
1401 }
1402
1403 // service is numeric due to NI_NUMERICSERV; parse without locale.
1404 const auto raw = std::strtoul(servBuf, nullptr, 10);
1405 if (raw > static_cast<unsigned long>((std::numeric_limits<Port>::max)()))
1406 throw SocketException("Port out of range for Port type.");
1407
1408 host.assign(hostBuf);
1409 port = static_cast<Port>(raw);
