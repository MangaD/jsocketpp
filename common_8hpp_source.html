798[[nodiscard]] inline AddrinfoPtr resolveAddress(const std::string_view host, const Port port, const int family,
799 const int socktype, const int protocol, const int flags = 0)
800{
801 addrinfo hints{};
802 hints.ai_family = family;
803 hints.ai_socktype = socktype;
804 hints.ai_protocol = protocol;
805 hints.ai_flags = flags;
806
807 const std::string portStr = std::to_string(port);
808 addrinfo* raw = nullptr;
809
810 if (const int ret = ::getaddrinfo(host.empty() ? nullptr : host.data(), portStr.c_str(), &hints, &raw); ret != 0)
811 {
812 throw SocketException(
813#ifdef _WIN32
814 GetSocketError(), SocketErrorMessageWrap(GetSocketError(), true)
815#else
816 ret, SocketErrorMessageWrap(ret, true)
817#endif
818 );
819 }
820
821 return AddrinfoPtr{raw};
822}
