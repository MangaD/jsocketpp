742 template <typename T> [[nodiscard]] T read()
743 {
744 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
745 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
746
747 std::array<std::byte, sizeof(T)> buffer{};
748 std::size_t totalRead = 0;
749 std::size_t remaining = sizeof(T);
750
751 while (remaining > 0)
752 {
753 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
754#ifdef _WIN32
755 static_cast<int>(remaining),
756#else
757 remaining,
758#endif
759 0);
760
761 if (len == SOCKET_ERROR)
762 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
763
764 if (len == 0)
765 throw SocketException("Connection closed by remote host before full object was received.");
766
767 totalRead += static_cast<std::size_t>(len);
768 remaining -= static_cast<std::size_t>(len);
769 }
770
771 return std::bit_cast<T>(buffer);
772 }

1905 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1906 {
1907 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1908 "Prefix type must be a trivially copyable integral type");
1909
1910 const std::size_t payloadSize = payload.size();
1911
1912 // Guard against overflows: ensure payload fits in the prefix type
1913 // Note: Put parentheses around the `max` function name due to a collision with
1914 // a macro defined in windows.h. See:
1915 // https://stackoverflow.com/a/13420838/3049315
1916 // Another alternative would be to define `NOMINMAX` before including windows.h
1917 // such as in CMake when compiling the target library
1918 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1919 {
1920 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
1921 " exceeds maximum encodable size for prefix type");
1922 }
1923
1924 // Convert the length to network byte order
1925 T len = net::toNetwork(static_cast<T>(payloadSize));
1926
1927 // Copy the prefix value into a properly aligned buffer to avoid UB
1928 std::array<char, sizeof(T)> prefixBuffer{};
1929 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1930 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1931
1932 // Write the length prefix
1933 const std::size_t prefixSent = writeAll(prefixView);
1934
1935 // Then write the payload
1936 const std::size_t dataSent = writeAll(payload);
1937
1938 return prefixSent + dataSent;
1939 }

1993 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1994 {
1995 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1996 "Prefix type must be a trivially copyable integral type");
1997
1998 // Check for null pointer with non-zero length
1999 if (!data && len > 0)
2000 {
2001 throw SocketException("writePrefixed: Null data pointer with non-zero length");
2002 }
2003
2004 // Ensure the payload length fits in the prefix type
2005 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2006 {
2007 throw SocketException("writePrefixed: Payload length " + std::to_string(len) +
2008 " exceeds capacity of prefix type (" +
2009 std::to_string((std::numeric_limits<T>::max)()) + ")");
2010 }
2011
2012 // Encode prefix in network byte order
2013 T prefix = net::toNetwork(static_cast<T>(len));
2014
2015 // Use aligned buffer to avoid undefined behavior when casting to char*
2016 std::array<char, sizeof(T)> prefixBuffer{};
2017 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
2018 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2019
2020 // Send prefix
2021 const std::size_t prefixSent = writeAll(prefixView);
2022
2023 // Send payload if any
2024 std::size_t payloadSent = 0;
2025 if (len > 0)
2026 {
2027 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
2028 }
2029
2030 return prefixSent + payloadSent;
2031 }
