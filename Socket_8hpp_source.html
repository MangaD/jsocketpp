744 template <typename T> [[nodiscard]] T read()
745 {
746 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
747 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
748
749 std::array<std::byte, sizeof(T)> buffer{};
750 std::size_t totalRead = 0;
751 std::size_t remaining = sizeof(T);
752
753 while (remaining > 0)
754 {
755 const auto len = ::recv(getSocketFd(), reinterpret_cast<char*>(buffer.data()) + totalRead,
756#ifdef _WIN32
757 static_cast<int>(remaining),
758#else
759 remaining,
760#endif
761 0);
762
763 if (len == SOCKET_ERROR)
764 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
765
766 if (len == 0)
767 throw SocketException("Connection closed by remote host before full object was received.");
768
769 totalRead += static_cast<std::size_t>(len);
770 remaining -= static_cast<std::size_t>(len);
771 }
772
773 return std::bit_cast<T>(buffer);
774 }

1907 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1908 {
1909 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1910 "Prefix type must be a trivially copyable integral type");
1911
1912 const std::size_t payloadSize = payload.size();
1913
1914 // Guard against overflows: ensure payload fits in the prefix type
1915 // Note: Put parentheses around the `max` function name due to a collision with
1916 // a macro defined in windows.h. See:
1917 // https://stackoverflow.com/a/13420838/3049315
1918 // Another alternative would be to define `NOMINMAX` before including windows.h
1919 // such as in CMake when compiling the target library
1920 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1921 {
1922 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
1923 " exceeds maximum encodable size for prefix type");
1924 }
1925
1926 // Convert the length to network byte order
1927 T len = net::toNetwork(static_cast<T>(payloadSize));
1928
1929 // Copy the prefix value into a properly aligned buffer to avoid UB
1930 std::array<char, sizeof(T)> prefixBuffer{};
1931 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1932 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1933
1934 // Write the length prefix
1935 const std::size_t prefixSent = writeAll(prefixView);
1936
1937 // Then write the payload
1938 const std::size_t dataSent = writeAll(payload);
1939
1940 return prefixSent + dataSent;
1941 }

1995 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1996 {
1997 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1998 "Prefix type must be a trivially copyable integral type");
1999
2000 // Check for null pointer with non-zero length
2001 if (!data && len > 0)
2002 {
2003 throw SocketException("writePrefixed: Null data pointer with non-zero length");
2004 }
2005
2006 // Ensure the payload length fits in the prefix type
2007 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2008 {
2009 throw SocketException("writePrefixed: Payload length " + std::to_string(len) +
2010 " exceeds capacity of prefix type (" +
2011 std::to_string((std::numeric_limits<T>::max)()) + ")");
2012 }
2013
2014 // Encode prefix in network byte order
2015 T prefix = net::toNetwork(static_cast<T>(len));
2016
2017 // Use aligned buffer to avoid undefined behavior when casting to char*
2018 std::array<char, sizeof(T)> prefixBuffer{};
2019 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
2020 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2021
2022 // Send prefix
2023 const std::size_t prefixSent = writeAll(prefixView);
2024
2025 // Send payload if any
2026 std::size_t payloadSent = 0;
2027 if (len > 0)
2028 {
2029 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
2030 }
2031
2032 return prefixSent + payloadSent;
2033 }
