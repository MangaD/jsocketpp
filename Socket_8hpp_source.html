544 template <typename T> T read()
545 {
546 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
547 T r;
548 std::array<char, sizeof(T)> buffer;
549 size_t totalRead = 0;
550 size_t remaining = sizeof(T);
551
552 while (remaining > 0)
553 {
554 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
555#ifdef _WIN32
556 static_cast<int>(remaining),
557#else
558 remaining,
559#endif
560 0);
561 if (len == SOCKET_ERROR)
562 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
563 if (len == 0)
564 throw SocketException(0, "Connection closed by remote host.");
565
566 totalRead += len;
567 remaining -= len;
568 }
569 std::memcpy(&r, buffer.data(), sizeof(T));
570 return r;
571 }

1650 template <typename T> std::size_t writePrefixed(const std::string& payload) const
1651 {
1652 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1653 "Prefix type must be a trivially copyable integral type");
1654
1655 const std::size_t payloadSize = payload.size();
1656
1657 // Put parentheses around the `max` function name due to a collision with
1658 // a macro defined in windows.h. See:
1659 // https://stackoverflow.com/a/13420838/3049315
1660 // Another alternative would be to define `NOMINMAX` before including windows.h
1661 // such as in CMake when compiling the target library
1662 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1663 {
1664 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1665 " exceeds maximum encodable size for prefix type");
1666 }
1667
1668 T len = static_cast<T>(payloadSize);
1669
1670 // Write the length prefix
1671 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1672
1673 // Then write the payload
1674 const std::size_t dataSent = writeAll(payload);
1675
1676 return prefixSent + dataSent;
1677 }

1727 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1728 {
1729 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1730 "Prefix type must be a trivially copyable integral type");
1731
1732 if (!data && len > 0)
1733 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1734
1735 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1736 {
1737 throw SocketException(0, "writePrefixed: Payload length exceeds capacity of prefix type");
1738 }
1739
1740 T prefix = static_cast<T>(len);
1741
1742 // Send prefix
1743 writeAll(std::string_view(reinterpret_cast<const char*>(&prefix), sizeof(T)));
1744
1745 // Send payload
1746 if (len > 0)
1747 {
1748 writeAll(std::string_view(static_cast<const char*>(data), len));
1749 }
1750
1751 return sizeof(T) + len;
1752 }
