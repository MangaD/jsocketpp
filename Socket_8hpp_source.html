622 template <typename T> [[nodiscard]] T read()
623 {
624 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
625 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
626
627 std::array<std::byte, sizeof(T)> buffer{};
628 std::size_t totalRead = 0;
629 std::size_t remaining = sizeof(T);
630
631 while (remaining > 0)
632 {
633 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
634#ifdef _WIN32
635 static_cast<int>(remaining),
636#else
637 remaining,
638#endif
639 0);
640
641 if (len == SOCKET_ERROR)
642 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
643
644 if (len == 0)
645 throw SocketException(0, "Connection closed by remote host before full object was received.");
646
647 totalRead += static_cast<std::size_t>(len);
648 remaining -= static_cast<std::size_t>(len);
649 }
650
651 return std::bit_cast<T>(buffer);
652 }

1785 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1786 {
1787 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1788 "Prefix type must be a trivially copyable integral type");
1789
1790 const std::size_t payloadSize = payload.size();
1791
1792 // Guard against overflows: ensure payload fits in the prefix type
1793 // Note: Put parentheses around the `max` function name due to a collision with
1794 // a macro defined in windows.h. See:
1795 // https://stackoverflow.com/a/13420838/3049315
1796 // Another alternative would be to define `NOMINMAX` before including windows.h
1797 // such as in CMake when compiling the target library
1798 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1799 {
1800 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1801 " exceeds maximum encodable size for prefix type");
1802 }
1803
1804 // Convert the length to network byte order
1805 T len = net::toNetwork(static_cast<T>(payloadSize));
1806
1807 // Copy the prefix value into a properly aligned buffer to avoid UB
1808 std::array<char, sizeof(T)> prefixBuffer{};
1809 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1810 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1811
1812 // Write the length prefix
1813 const std::size_t prefixSent = writeAll(prefixView);
1814
1815 // Then write the payload
1816 const std::size_t dataSent = writeAll(payload);
1817
1818 return prefixSent + dataSent;
1819 }

1873 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1874 {
1875 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1876 "Prefix type must be a trivially copyable integral type");
1877
1878 // Check for null pointer with non-zero length
1879 if (!data && len > 0)
1880 {
1881 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1882 }
1883
1884 // Ensure the payload length fits in the prefix type
1885 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1886 {
1887 throw SocketException(0, "writePrefixed: Payload length " + std::to_string(len) +
1888 " exceeds capacity of prefix type (" +
1889 std::to_string((std::numeric_limits<T>::max)()) + ")");
1890 }
1891
1892 // Encode prefix in network byte order
1893 T prefix = net::toNetwork(static_cast<T>(len));
1894
1895 // Use aligned buffer to avoid undefined behavior when casting to char*
1896 std::array<char, sizeof(T)> prefixBuffer{};
1897 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
1898 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1899
1900 // Send prefix
1901 const std::size_t prefixSent = writeAll(prefixView);
1902
1903 // Send payload if any
1904 std::size_t payloadSent = 0;
1905 if (len > 0)
1906 {
1907 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
1908 }
1909
1910 return prefixSent + payloadSent;
