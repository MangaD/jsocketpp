529 template <typename T> T read()
530 {
531 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
532 T r;
533 std::array<char, sizeof(T)> buffer;
534 size_t totalRead = 0;
535 size_t remaining = sizeof(T);
536
537 while (remaining > 0)
538 {
539 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
540#ifdef _WIN32
541 static_cast<int>(remaining),
542#else
543 remaining,
544#endif
545 0);
546 if (len == SOCKET_ERROR)
547 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
548 if (len == 0)
549 throw SocketException(0, "Connection closed by remote host.");
550
551 totalRead += len;
552 remaining -= len;
553 }
554 std::memcpy(&r, buffer.data(), sizeof(T));
555 return r;
556 }

1320 template <typename T> std::size_t writePrefixed(const std::string& payload) const
1321 {
1322 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1323 "Prefix type must be a trivially copyable integral type");
1324
1325 const std::size_t payloadSize = payload.size();
1326
1327 // Put parentheses around the `max` function name due to a collision with
1328 // a macro defined in windows.h. See:
1329 // https://stackoverflow.com/a/13420838/3049315
1330 // Another alternative would be to define `NOMINMAX` before including windows.h
1331 // such as in CMake when compiling the target library
1332 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1333 {
1334 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1335 " exceeds maximum encodable size for prefix type");
1336 }
1337
1338 T len = static_cast<T>(payloadSize);
1339
1340 // Write the length prefix
1341 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1342
1343 // Then write the payload
1344 const std::size_t dataSent = writeAll(payload);
1345
1346 return prefixSent + dataSent;
1347 }
