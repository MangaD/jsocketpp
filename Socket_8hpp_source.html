530 template <typename T> T read()
531 {
532 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
533 T r;
534 std::array<char, sizeof(T)> buffer;
535 size_t totalRead = 0;
536 size_t remaining = sizeof(T);
537
538 while (remaining > 0)
539 {
540 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
541#ifdef _WIN32
542 static_cast<int>(remaining),
543#else
544 remaining,
545#endif
546 0);
547 if (len == SOCKET_ERROR)
548 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
549 if (len == 0)
550 throw SocketException(0, "Connection closed by remote host.");
551
552 totalRead += len;
553 remaining -= len;
554 }
555 std::memcpy(&r, buffer.data(), sizeof(T));

1495 template <typename T> std::size_t writePrefixed(const std::string& payload) const
1496 {
1497 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1498 "Prefix type must be a trivially copyable integral type");
1499
1500 const std::size_t payloadSize = payload.size();
1501
1502 // Put parentheses around the `max` function name due to a collision with
1503 // a macro defined in windows.h. See:
1504 // https://stackoverflow.com/a/13420838/3049315
1505 // Another alternative would be to define `NOMINMAX` before including windows.h
1506 // such as in CMake when compiling the target library
1507 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1508 {
1509 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1510 " exceeds maximum encodable size for prefix type");
1511 }
1512
1513 T len = static_cast<T>(payloadSize);
1514
1515 // Write the length prefix
1516 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1517
1518 // Then write the payload
1519 const std::size_t dataSent = writeAll(payload);
1520
1521 return prefixSent + dataSent;
1522 }

1572 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1573 {
1574 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1575 "Prefix type must be a trivially copyable integral type");
1576
1577 if (!data && len > 0)
1578 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1579
1580 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1581 {
1582 throw SocketException(0, "writePrefixed: Payload length exceeds capacity of prefix type");
1583 }
1584
1585 T prefix = static_cast<T>(len);
1586
1587 // Send prefix
1588 writeAll(std::string_view(reinterpret_cast<const char*>(&prefix), sizeof(T)));
1589
1590 // Send payload
1591 if (len > 0)
1592 {
1593 writeAll(std::string_view(static_cast<const char*>(data), len));
1594 }
1595
1596 return sizeof(T) + len;
