548 template <typename T> T read()
549 {
550 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
551 T r;
552 std::array<char, sizeof(T)> buffer;
553 size_t totalRead = 0;
554 size_t remaining = sizeof(T);
555
556 while (remaining > 0)
557 {
558 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
559#ifdef _WIN32
560 static_cast<int>(remaining),
561#else
562 remaining,
563#endif
564 0);
565 if (len == SOCKET_ERROR)
566 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
567 if (len == 0)
568 throw SocketException(0, "Connection closed by remote host.");
569
570 totalRead += len;
571 remaining -= len;
572 }
573 std::memcpy(&r, buffer.data(), sizeof(T));
574 return r;
575 }

1665 template <typename T> std::size_t writePrefixed(const std::string& payload)
1666 {
1667 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1668 "Prefix type must be a trivially copyable integral type");
1669
1670 const std::size_t payloadSize = payload.size();
1671
1672 // Put parentheses around the `max` function name due to a collision with
1673 // a macro defined in windows.h. See:
1674 // https://stackoverflow.com/a/13420838/3049315
1675 // Another alternative would be to define `NOMINMAX` before including windows.h
1676 // such as in CMake when compiling the target library
1677 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1678 {
1679 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1680 " exceeds maximum encodable size for prefix type");
1681 }
1682
1683 T len = net::toNetwork(static_cast<T>(payloadSize));
1684
1685 // Write the length prefix
1686 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1687
1688 // Then write the payload
1689 const std::size_t dataSent = writeAll(payload);
1690
1691 return prefixSent + dataSent;
1692 }

1745 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1746 {
1747 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1748 "Prefix type must be a trivially copyable integral type");
1749
1750 if (!data && len > 0)
1751 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1752
1753 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1754 {
1755 throw SocketException(0, "writePrefixed: Payload length exceeds capacity of prefix type");
1756 }
1757
1758 T prefix = net::toNetwork(static_cast<T>(len));
1759
1760 // Send prefix
1761 writeAll(std::string_view(reinterpret_cast<const char*>(&prefix), sizeof(T)));
1762
1763 // Send payload
1764 if (len > 0)
1765 {
1766 writeAll(std::string_view(static_cast<const char*>(data), len));
1767 }
1768
1769 return sizeof(T) + len;
1770 }
