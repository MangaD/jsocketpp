929 {
930 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
931 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");

934 std::size_t totalRead = 0;
935 std::size_t remaining = sizeof(T);

937 while (remaining > 0)
938 {

940#ifdef _WIN32
941 static_cast<int>(remaining),
942#else
943 remaining,
944#endif

946

948 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));

2091 template <typename T> std::size_t writePrefixed(const std::string_view payload)
2092 {
2093 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
2094 "Prefix type must be a trivially copyable integral type");
2095
2096 const std::size_t payloadSize = payload.size();
2097
2098 // Guard against overflows: ensure payload fits in the prefix type
2099 // Note: Put parentheses around the `max` function name due to a collision with
2100 // a macro defined in windows.h. See:
2101 // https://stackoverflow.com/a/13420838/3049315
2102 // Another alternative would be to define `NOMINMAX` before including windows.h
2103 // such as in CMake when compiling the target library
2104 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2105 {
2106 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
2107 " exceeds maximum encodable size for prefix type");
2108 }
2109
2110 // Convert the length to network byte order
2111 T len = net::toNetwork(static_cast<T>(payloadSize));
2112
2113 // Copy the prefix value into a properly aligned buffer to avoid UB
2114 std::array<char, sizeof(T)> prefixBuffer{};
2115 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
2116 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2117
2118 // Write the length prefix
2119 const std::size_t prefixSent = writeAll(prefixView);
2120
2121 // Then write the payload
2122 const std::size_t dataSent = writeAll(payload);
2123
2124 return prefixSent + dataSent;
