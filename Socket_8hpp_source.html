737 template <typename T> [[nodiscard]] T read()
738 {
739 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
740 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
741
742 std::array<std::byte, sizeof(T)> buffer{};
743 std::size_t totalRead = 0;
744 std::size_t remaining = sizeof(T);
745
746 while (remaining > 0)
747 {
748 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
749#ifdef _WIN32
750 static_cast<int>(remaining),
751#else
752 remaining,
753#endif
754 0);
755
756 if (len == SOCKET_ERROR)
757 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
758
759 if (len == 0)
760 throw SocketException("Connection closed by remote host before full object was received.");
761
762 totalRead += static_cast<std::size_t>(len);
763 remaining -= static_cast<std::size_t>(len);
764 }
765
766 return std::bit_cast<T>(buffer);
767 }

1900 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1901 {
1902 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1903 "Prefix type must be a trivially copyable integral type");
1904
1905 const std::size_t payloadSize = payload.size();
1906
1907 // Guard against overflows: ensure payload fits in the prefix type
1908 // Note: Put parentheses around the `max` function name due to a collision with
1909 // a macro defined in windows.h. See:
1910 // https://stackoverflow.com/a/13420838/3049315
1911 // Another alternative would be to define `NOMINMAX` before including windows.h
1912 // such as in CMake when compiling the target library
1913 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1914 {
1915 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
1916 " exceeds maximum encodable size for prefix type");
1917 }
1918
1919 // Convert the length to network byte order
1920 T len = net::toNetwork(static_cast<T>(payloadSize));
1921
1922 // Copy the prefix value into a properly aligned buffer to avoid UB
1923 std::array<char, sizeof(T)> prefixBuffer{};
1924 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1925 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1926
1927 // Write the length prefix
1928 const std::size_t prefixSent = writeAll(prefixView);
1929
1930 // Then write the payload
1931 const std::size_t dataSent = writeAll(payload);
1932
1933 return prefixSent + dataSent;
1934 }

1988 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1989 {
1990 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1991 "Prefix type must be a trivially copyable integral type");
1992
1993 // Check for null pointer with non-zero length
1994 if (!data && len > 0)
1995 {
1996 throw SocketException("writePrefixed: Null data pointer with non-zero length");
1997 }
1998
1999 // Ensure the payload length fits in the prefix type
2000 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2001 {
2002 throw SocketException("writePrefixed: Payload length " + std::to_string(len) +
2003 " exceeds capacity of prefix type (" +
2004 std::to_string((std::numeric_limits<T>::max)()) + ")");
2005 }
2006
2007 // Encode prefix in network byte order
2008 T prefix = net::toNetwork(static_cast<T>(len));
2009
2010 // Use aligned buffer to avoid undefined behavior when casting to char*
2011 std::array<char, sizeof(T)> prefixBuffer{};
2012 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
2013 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2014
2015 // Send prefix
2016 const std::size_t prefixSent = writeAll(prefixView);
2017
2018 // Send payload if any
2019 std::size_t payloadSent = 0;
2020 if (len > 0)
2021 {
2022 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
2023 }
2024
2025 return prefixSent + payloadSent;
2026 }
