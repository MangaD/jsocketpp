627 template <typename T> [[nodiscard]] T read()
628 {
629 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
630 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
631
632 std::array<std::byte, sizeof(T)> buffer{};
633 std::size_t totalRead = 0;
634 std::size_t remaining = sizeof(T);
635
636 while (remaining > 0)
637 {
638 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
639#ifdef _WIN32
640 static_cast<int>(remaining),
641#else
642 remaining,
643#endif
644 0);
645
646 if (len == SOCKET_ERROR)
647 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
648
649 if (len == 0)
650 throw SocketException("Connection closed by remote host before full object was received.");
651
652 totalRead += static_cast<std::size_t>(len);
653 remaining -= static_cast<std::size_t>(len);
654 }
655
656 return std::bit_cast<T>(buffer);
657 }

1790 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1791 {
1792 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1793 "Prefix type must be a trivially copyable integral type");
1794
1795 const std::size_t payloadSize = payload.size();
1796
1797 // Guard against overflows: ensure payload fits in the prefix type
1798 // Note: Put parentheses around the `max` function name due to a collision with
1799 // a macro defined in windows.h. See:
1800 // https://stackoverflow.com/a/13420838/3049315
1801 // Another alternative would be to define `NOMINMAX` before including windows.h
1802 // such as in CMake when compiling the target library
1803 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1804 {
1805 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
1806 " exceeds maximum encodable size for prefix type");
1807 }
1808
1809 // Convert the length to network byte order
1810 T len = net::toNetwork(static_cast<T>(payloadSize));
1811
1812 // Copy the prefix value into a properly aligned buffer to avoid UB
1813 std::array<char, sizeof(T)> prefixBuffer{};
1814 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1815 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1816
1817 // Write the length prefix
1818 const std::size_t prefixSent = writeAll(prefixView);
1819
1820 // Then write the payload
1821 const std::size_t dataSent = writeAll(payload);
1822
1823 return prefixSent + dataSent;
1824 }

1878 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1879 {
1880 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1881 "Prefix type must be a trivially copyable integral type");
1882
1883 // Check for null pointer with non-zero length
1884 if (!data && len > 0)
1885 {
1886 throw SocketException("writePrefixed: Null data pointer with non-zero length");
1887 }
1888
1889 // Ensure the payload length fits in the prefix type
1890 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1891 {
1892 throw SocketException("writePrefixed: Payload length " + std::to_string(len) +
1893 " exceeds capacity of prefix type (" +
1894 std::to_string((std::numeric_limits<T>::max)()) + ")");
1895 }
1896
1897 // Encode prefix in network byte order
1898 T prefix = net::toNetwork(static_cast<T>(len));
1899
1900 // Use aligned buffer to avoid undefined behavior when casting to char*
1901 std::array<char, sizeof(T)> prefixBuffer{};
1902 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
1903 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1904
1905 // Send prefix
1906 const std::size_t prefixSent = writeAll(prefixView);
1907
1908 // Send payload if any
1909 std::size_t payloadSent = 0;
1910 if (len > 0)
1911 {
1912 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
1913 }
1914
1915 return prefixSent + payloadSent;
1916 }
