567 template <typename T> T read()
568 {
569 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
570 T r;
571 std::array<char, sizeof(T)> buffer;
572 size_t totalRead = 0;
573 size_t remaining = sizeof(T);
574
575 while (remaining > 0)
576 {
577 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
578#ifdef _WIN32
579 static_cast<int>(remaining),
580#else
581 remaining,
582#endif
583 0);
584 if (len == SOCKET_ERROR)
585 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
586 if (len == 0)
587 throw SocketException(0, "Connection closed by remote host.");
588
589 totalRead += len;
590 remaining -= len;
591 }
592 std::memcpy(&r, buffer.data(), sizeof(T));
593 return r;
594 }

1673 template <typename T> std::size_t writePrefixed(const std::string& payload) const
1674 {
1675 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1676 "Prefix type must be a trivially copyable integral type");
1677
1678 const std::size_t payloadSize = payload.size();
1679
1680 // Put parentheses around the `max` function name due to a collision with
1681 // a macro defined in windows.h. See:
1682 // https://stackoverflow.com/a/13420838/3049315
1683 // Another alternative would be to define `NOMINMAX` before including windows.h
1684 // such as in CMake when compiling the target library
1685 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1686 {
1687 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1688 " exceeds maximum encodable size for prefix type");
1689 }
1690
1691 T len = static_cast<T>(payloadSize);
1692
1693 // Write the length prefix
1694 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1695
1696 // Then write the payload
1697 const std::size_t dataSent = writeAll(payload);
1698
1699 return prefixSent + dataSent;

1750 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1751 {
1752 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1753 "Prefix type must be a trivially copyable integral type");
1754
1755 if (!data && len > 0)
1756 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1757
1758 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1759 {
1760 throw SocketException(0, "writePrefixed: Payload length exceeds capacity of prefix type");
1761 }
1762
1763 T prefix = static_cast<T>(len);
1764
1765 // Send prefix
1766 writeAll(std::string_view(reinterpret_cast<const char*>(&prefix), sizeof(T)));
1767
1768 // Send payload
1769 if (len > 0)
1770 {
1771 writeAll(std::string_view(static_cast<const char*>(data), len));
1772 }
1773
1774 return sizeof(T) + len;
