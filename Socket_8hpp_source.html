878 {
879 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
880 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");

883 std::size_t totalRead = 0;

889#ifdef _WIN32

2040 template <typename T> std::size_t writePrefixed(const std::string_view payload)
2041 {
2042 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
2043 "Prefix type must be a trivially copyable integral type");
2044
2045 const std::size_t payloadSize = payload.size();
2046
2047 // Guard against overflows: ensure payload fits in the prefix type
2048 // Note: Put parentheses around the `max` function name due to a collision with
2049 // a macro defined in windows.h. See:
2050 // https://stackoverflow.com/a/13420838/3049315
2051 // Another alternative would be to define `NOMINMAX` before including windows.h
2052 // such as in CMake when compiling the target library
2053 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2054 {
2055 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
2056 " exceeds maximum encodable size for prefix type");
2057 }
2058
2059 // Convert the length to network byte order
2060 T len = net::toNetwork(static_cast<T>(payloadSize));
2061
2062 // Copy the prefix value into a properly aligned buffer to avoid UB
2063 std::array<char, sizeof(T)> prefixBuffer{};
2064 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
2065 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2066
2067 // Write the length prefix
2068 const std::size_t prefixSent = writeAll(prefixView);
2069
2070 // Then write the payload
2071 const std::size_t dataSent = writeAll(payload);
2072
2073 return prefixSent + dataSent;
