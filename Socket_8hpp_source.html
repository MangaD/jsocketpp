555 template <typename T> [[nodiscard]] T read()
556 {
557 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
558 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
559
560 std::array<std::byte, sizeof(T)> buffer{};
561 std::size_t totalRead = 0;
562 std::size_t remaining = sizeof(T);
563
564 while (remaining > 0)
565 {
566 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
567#ifdef _WIN32
568 static_cast<int>(remaining),
569#else
570 remaining,
571#endif
572 0);
573
574 if (len == SOCKET_ERROR)
575 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
576
577 if (len == 0)
578 throw SocketException(0, "Connection closed by remote host before full object was received.");
579
580 totalRead += static_cast<std::size_t>(len);
581 remaining -= static_cast<std::size_t>(len);
582 }
583
584 return std::bit_cast<T>(buffer);
585 }

1718 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1719 {
1720 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1721 "Prefix type must be a trivially copyable integral type");
1722
1723 const std::size_t payloadSize = payload.size();
1724
1725 // Guard against overflows: ensure payload fits in the prefix type
1726 // Note: Put parentheses around the `max` function name due to a collision with
1727 // a macro defined in windows.h. See:
1728 // https://stackoverflow.com/a/13420838/3049315
1729 // Another alternative would be to define `NOMINMAX` before including windows.h
1730 // such as in CMake when compiling the target library
1731 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1732 {
1733 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1734 " exceeds maximum encodable size for prefix type");
1735 }
1736
1737 // Convert the length to network byte order
1738 T len = net::toNetwork(static_cast<T>(payloadSize));
1739
1740 // Copy the prefix value into a properly aligned buffer to avoid UB
1741 std::array<char, sizeof(T)> prefixBuffer{};
1742 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1743 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1744
1745 // Write the length prefix
1746 const std::size_t prefixSent = writeAll(prefixView);
1747
1748 // Then write the payload
1749 const std::size_t dataSent = writeAll(payload);
1750
1751 return prefixSent + dataSent;
1752 }

1806 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1807 {
1808 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1809 "Prefix type must be a trivially copyable integral type");
1810
1811 // Check for null pointer with non-zero length
1812 if (!data && len > 0)
1813 {
1814 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1815 }
1816
1817 // Ensure the payload length fits in the prefix type
1818 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1819 {
1820 throw SocketException(0, "writePrefixed: Payload length " + std::to_string(len) +
1821 " exceeds capacity of prefix type (" +
1822 std::to_string((std::numeric_limits<T>::max)()) + ")");
1823 }
1824
1825 // Encode prefix in network byte order
1826 T prefix = net::toNetwork(static_cast<T>(len));
1827
1828 // Use aligned buffer to avoid undefined behavior when casting to char*
1829 std::array<char, sizeof(T)> prefixBuffer{};
1830 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
1831 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1832
1833 // Send prefix
1834 const std::size_t prefixSent = writeAll(prefixView);
1835
1836 // Send payload if any
1837 std::size_t payloadSent = 0;
1838 if (len > 0)
1839 {
1840 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
1841 }
1842
1843 return prefixSent + payloadSent;
