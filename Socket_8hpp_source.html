753 template <typename T> [[nodiscard]] T read()
754 {
755 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
756 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
757
758 std::array<std::byte, sizeof(T)> buffer{};
759 std::size_t totalRead = 0;
760 std::size_t remaining = sizeof(T);
761
762 while (remaining > 0)
763 {
764 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
765#ifdef _WIN32
766 static_cast<int>(remaining),
767#else
768 remaining,
769#endif
770 0);
771
772 if (len == SOCKET_ERROR)
773 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
774
775 if (len == 0)
776 throw SocketException("Connection closed by remote host before full object was received.");
777
778 totalRead += static_cast<std::size_t>(len);
779 remaining -= static_cast<std::size_t>(len);
780 }
781
782 return std::bit_cast<T>(buffer);
783 }

1916 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1917 {
1918 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1919 "Prefix type must be a trivially copyable integral type");
1920
1921 const std::size_t payloadSize = payload.size();
1922
1923 // Guard against overflows: ensure payload fits in the prefix type
1924 // Note: Put parentheses around the `max` function name due to a collision with
1925 // a macro defined in windows.h. See:
1926 // https://stackoverflow.com/a/13420838/3049315
1927 // Another alternative would be to define `NOMINMAX` before including windows.h
1928 // such as in CMake when compiling the target library
1929 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1930 {
1931 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
1932 " exceeds maximum encodable size for prefix type");
1933 }
1934
1935 // Convert the length to network byte order
1936 T len = net::toNetwork(static_cast<T>(payloadSize));
1937
1938 // Copy the prefix value into a properly aligned buffer to avoid UB
1939 std::array<char, sizeof(T)> prefixBuffer{};
1940 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1941 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1942
1943 // Write the length prefix
1944 const std::size_t prefixSent = writeAll(prefixView);
1945
1946 // Then write the payload
1947 const std::size_t dataSent = writeAll(payload);
1948
1949 return prefixSent + dataSent;
1950 }

2004 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
2005 {
2006 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
2007 "Prefix type must be a trivially copyable integral type");
2008
2009 // Check for null pointer with non-zero length
2010 if (!data && len > 0)
2011 {
2012 throw SocketException("writePrefixed: Null data pointer with non-zero length");
2013 }
2014
2015 // Ensure the payload length fits in the prefix type
2016 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2017 {
2018 throw SocketException("writePrefixed: Payload length " + std::to_string(len) +
2019 " exceeds capacity of prefix type (" +
2020 std::to_string((std::numeric_limits<T>::max)()) + ")");
2021 }
2022
2023 // Encode prefix in network byte order
2024 T prefix = net::toNetwork(static_cast<T>(len));
2025
2026 // Use aligned buffer to avoid undefined behavior when casting to char*
2027 std::array<char, sizeof(T)> prefixBuffer{};
2028 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
2029 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2030
2031 // Send prefix
2032 const std::size_t prefixSent = writeAll(prefixView);
2033
2034 // Send payload if any
2035 std::size_t payloadSent = 0;
2036 if (len > 0)
2037 {
2038 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
2039 }
2040
2041 return prefixSent + payloadSent;
2042 }
