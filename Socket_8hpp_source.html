553 template <typename T> T read()
554 {
555 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
556 T r;
557 std::array<char, sizeof(T)> buffer;
558 size_t totalRead = 0;
559 size_t remaining = sizeof(T);
560
561 while (remaining > 0)
562 {
563 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
564#ifdef _WIN32
565 static_cast<int>(remaining),
566#else
567 remaining,
568#endif
569 0);
570 if (len == SOCKET_ERROR)
571 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
572 if (len == 0)
573 throw SocketException(0, "Connection closed by remote host.");
574
575 totalRead += len;
576 remaining -= len;
577 }
578 std::memcpy(&r, buffer.data(), sizeof(T));
579 return r;
580 }

1670 template <typename T> std::size_t writePrefixed(const std::string& payload)
1671 {
1672 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1673 "Prefix type must be a trivially copyable integral type");
1674
1675 const std::size_t payloadSize = payload.size();
1676
1677 // Put parentheses around the `max` function name due to a collision with
1678 // a macro defined in windows.h. See:
1679 // https://stackoverflow.com/a/13420838/3049315
1680 // Another alternative would be to define `NOMINMAX` before including windows.h
1681 // such as in CMake when compiling the target library
1682 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1683 {
1684 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1685 " exceeds maximum encodable size for prefix type");
1686 }
1687
1688 T len = net::toNetwork(static_cast<T>(payloadSize));
1689
1690 // Write the length prefix
1691 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1692
1693 // Then write the payload
1694 const std::size_t dataSent = writeAll(payload);
1695
1696 return prefixSent + dataSent;

1750 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1751 {
1752 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1753 "Prefix type must be a trivially copyable integral type");
1754
1755 if (!data && len > 0)
1756 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1757
1758 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1759 {
1760 throw SocketException(0, "writePrefixed: Payload length exceeds capacity of prefix type");
1761 }
1762
1763 T prefix = net::toNetwork(static_cast<T>(len));
1764
1765 // Send prefix
1766 writeAll(std::string_view(reinterpret_cast<const char*>(&prefix), sizeof(T)));
1767
1768 // Send payload
1769 if (len > 0)
1770 {
1771 writeAll(std::string_view(static_cast<const char*>(data), len));
1772 }
1773
1774 return sizeof(T) + len;
1775 }
