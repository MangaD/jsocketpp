748 template <typename T> [[nodiscard]] T read()
749 {
750 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() requires a trivially copyable type");
751 static_assert(std::is_standard_layout_v<T>, "Socket::read<T>() requires a standard layout type");
752
753 std::array<std::byte, sizeof(T)> buffer{};
754 std::size_t totalRead = 0;
755 std::size_t remaining = sizeof(T);
756
757 while (remaining > 0)
758 {
759 const auto len = recv(_sockFd, reinterpret_cast<char*>(buffer.data()) + totalRead,
760#ifdef _WIN32
761 static_cast<int>(remaining),
762#else
763 remaining,
764#endif
765 0);
766
767 if (len == SOCKET_ERROR)
768 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
769
770 if (len == 0)
771 throw SocketException("Connection closed by remote host before full object was received.");
772
773 totalRead += static_cast<std::size_t>(len);
774 remaining -= static_cast<std::size_t>(len);
775 }
776
777 return std::bit_cast<T>(buffer);
778 }

1911 template <typename T> std::size_t writePrefixed(const std::string_view payload)
1912 {
1913 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1914 "Prefix type must be a trivially copyable integral type");
1915
1916 const std::size_t payloadSize = payload.size();
1917
1918 // Guard against overflows: ensure payload fits in the prefix type
1919 // Note: Put parentheses around the `max` function name due to a collision with
1920 // a macro defined in windows.h. See:
1921 // https://stackoverflow.com/a/13420838/3049315
1922 // Another alternative would be to define `NOMINMAX` before including windows.h
1923 // such as in CMake when compiling the target library
1924 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1925 {
1926 throw SocketException("writePrefixed: Payload size " + std::to_string(payloadSize) +
1927 " exceeds maximum encodable size for prefix type");
1928 }
1929
1930 // Convert the length to network byte order
1931 T len = net::toNetwork(static_cast<T>(payloadSize));
1932
1933 // Copy the prefix value into a properly aligned buffer to avoid UB
1934 std::array<char, sizeof(T)> prefixBuffer{};
1935 std::memcpy(prefixBuffer.data(), &len, sizeof(T));
1936 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
1937
1938 // Write the length prefix
1939 const std::size_t prefixSent = writeAll(prefixView);
1940
1941 // Then write the payload
1942 const std::size_t dataSent = writeAll(payload);
1943
1944 return prefixSent + dataSent;
1945 }

1999 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
2000 {
2001 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
2002 "Prefix type must be a trivially copyable integral type");
2003
2004 // Check for null pointer with non-zero length
2005 if (!data && len > 0)
2006 {
2007 throw SocketException("writePrefixed: Null data pointer with non-zero length");
2008 }
2009
2010 // Ensure the payload length fits in the prefix type
2011 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
2012 {
2013 throw SocketException("writePrefixed: Payload length " + std::to_string(len) +
2014 " exceeds capacity of prefix type (" +
2015 std::to_string((std::numeric_limits<T>::max)()) + ")");
2016 }
2017
2018 // Encode prefix in network byte order
2019 T prefix = net::toNetwork(static_cast<T>(len));
2020
2021 // Use aligned buffer to avoid undefined behavior when casting to char*
2022 std::array<char, sizeof(T)> prefixBuffer{};
2023 std::memcpy(prefixBuffer.data(), &prefix, sizeof(T));
2024 const std::string_view prefixView(prefixBuffer.data(), sizeof(T));
2025
2026 // Send prefix
2027 const std::size_t prefixSent = writeAll(prefixView);
2028
2029 // Send payload if any
2030 std::size_t payloadSent = 0;
2031 if (len > 0)
2032 {
2033 payloadSent = writeAll(std::string_view(static_cast<const char*>(data), len));
2034 }
2035
2036 return prefixSent + payloadSent;
2037 }
