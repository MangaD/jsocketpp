541 template <typename T> T read()
542 {
543 static_assert(std::is_trivially_copyable_v<T>, "Socket::read<T>() only supports trivially copyable types");
544 T r;
545 std::array<char, sizeof(T)> buffer;
546 size_t totalRead = 0;
547 size_t remaining = sizeof(T);
548
549 while (remaining > 0)
550 {
551 const auto len = recv(this->_sockFd, buffer.data() + totalRead,
552#ifdef _WIN32
553 static_cast<int>(remaining),
554#else
555 remaining,
556#endif
557 0);
558 if (len == SOCKET_ERROR)
559 throw SocketException(GetSocketError(), SocketErrorMessage(GetSocketError()));
560 if (len == 0)
561 throw SocketException(0, "Connection closed by remote host.");
562
563 totalRead += len;
564 remaining -= len;
565 }
566 std::memcpy(&r, buffer.data(), sizeof(T));
567 return r;
568 }

1658 template <typename T> std::size_t writePrefixed(const std::string& payload)
1659 {
1660 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1661 "Prefix type must be a trivially copyable integral type");
1662
1663 const std::size_t payloadSize = payload.size();
1664
1665 // Put parentheses around the `max` function name due to a collision with
1666 // a macro defined in windows.h. See:
1667 // https://stackoverflow.com/a/13420838/3049315
1668 // Another alternative would be to define `NOMINMAX` before including windows.h
1669 // such as in CMake when compiling the target library
1670 if (payloadSize > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1671 {
1672 throw SocketException(0, "writePrefixed: Payload size " + std::to_string(payloadSize) +
1673 " exceeds maximum encodable size for prefix type");
1674 }
1675
1676 T len = net::toNetwork(static_cast<T>(payloadSize));
1677
1678 // Write the length prefix
1679 const std::size_t prefixSent = writeAll(std::string_view(reinterpret_cast<const char*>(&len), sizeof(T)));
1680
1681 // Then write the payload
1682 const std::size_t dataSent = writeAll(payload);
1683
1684 return prefixSent + dataSent;
1685 }

1738 template <typename T> std::size_t writePrefixed(const void* data, std::size_t len) const
1739 {
1740 static_assert(std::is_integral_v<T> && std::is_trivially_copyable_v<T>,
1741 "Prefix type must be a trivially copyable integral type");
1742
1743 if (!data && len > 0)
1744 throw SocketException(0, "writePrefixed: Null data pointer with non-zero length");
1745
1746 if (len > static_cast<std::size_t>((std::numeric_limits<T>::max)()))
1747 {
1748 throw SocketException(0, "writePrefixed: Payload length exceeds capacity of prefix type");
1749 }
1750
1751 T prefix = net::toNetwork(static_cast<T>(len));
1752
1753 // Send prefix
1754 writeAll(std::string_view(reinterpret_cast<const char*>(&prefix), sizeof(T)));
1755
1756 // Send payload
1757 if (len > 0)
1758 {
1759 writeAll(std::string_view(static_cast<const char*>(data), len));
1760 }
1761
1762 return sizeof(T) + len;
